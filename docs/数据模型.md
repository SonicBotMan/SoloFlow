## 二、详细数据结构

### 2.1 老板 (Boss)

```typescript
interface Boss {
  id: string;                      // boss_001
  name: string;                    // 老板名字
  avatar?: string;
  
  // 创建时间
  created_at: string;
  
  // 偏好配置
  preferences: BossPreferences;
  
  // 活跃项目
  active_projects: string[];       // project_id[]
}
```

### 2.2 项目 (Project)

```typescript
interface Project {
  id: string;                      // proj_20260305_001
  boss_id: string;                 // 所属老板
  
  // 基本信息
  name: string;                    // 项目名称
  type: ProjectType;              // 项目类型
  status: ProjectStatus;
  
  // 需求
  original_requirement: string;   // 老板原始需求
  understood_requirement?: string; // 助理理解后的需求
  
  // 工作流
  workflow: WorkflowNode[];
  current_step: number;           // 当前步骤索引
  
  // 任务列表
  tasks: Task[];
  
  // 交付物
  deliverables: Deliverable[];
  
  // 时间
  created_at: string;
  updated_at: string;
  completed_at?: string;
  
  // 老板确认
  boss_approved: boolean;
  boss_comment?: string;
}

type ProjectType = 
  | 'hot_video'        // 热点视频
  | 'creative_video'  // 创意视频
  | 'product_video'   // 产品视频
  | 'image_series'    // 图片系列
  | 'other';          // 其他

type ProjectStatus = 
  | 'draft'           // 草稿（需求理解中）
  | 'planning'        // 规划中（任务拆解中）
  | 'executing'      // 执行中
  | 'reviewing'      // 待验收
  | 'completed'      // 已完成
  | 'rejected';      // 已驳回
```

### 2.3 任务 (Task)

```typescript
interface Task {
  id: string;                      // task_001
  project_id: string;
  
  // 执行者
  employee_role: EmployeeRole;
  employee_name: string;           // 员工花名
  
  // 任务信息
  name: string;                    // 任务名称
  description: string;             // 任务描述
  requirement: string;             // 任务要求
  
  // 状态
  status: TaskStatus;
  priority: 'low' | 'medium' | 'high';
  
  // 依赖
  depends_on: string[];            // 依赖的任务ID
  blocked_by: string[];           // 阻塞的任务ID
  
  // 执行信息
  assigned_at?: string;
  started_at?: string;
  completed_at?: string;
  
  // 结果
  result?: string;                 // 结果描述
  output_files?: string[];        // 输出文件路径
  
  // 反馈
  feedback?: TaskFeedback;
}

type TaskStatus = 
  | 'pending'           // 待开始
  | 'assigned'         // 已分配
  | 'in_progress'      // 进行中
  | 'completed'        // 已完成
  | 'waiting_review'  // 待验收
  | 'rejected'         // 已驳回
  | 'blocked';         // 被阻塞
```

### 2.4 员工 (Employee)

```typescript
type EmployeeRole = 
  | 'ideator'          // 点子王
  | 'copywriter'       // 文案师
  | 'material'         // 素材师
  | 'editor'           // 剪辑师
  | 'marketer'         // 营销专家
  | 'publisher';       // 发布专家

interface Employee {
  role: EmployeeRole;
  name: string;                    // 花名：小点/小文/小剪/小材/小营/小发
  description: string;
  
  // 能力
  capabilities: string[];
  skills: string[];                // 调用的Skill
  
  // 状态
  status: EmployeeStatus;
  current_task_id?: string;
  
  // 统计
  stats: {
    total_tasks: number;
    completed_tasks: number;
    rejected_count: number;
    avg_completion_time: number;  // 平均完成时间(分钟)
  };
  
  // 偏好理解
  preference_understanding: PreferenceUnderstanding;
  
  // 最近工作
  recent_tasks: TaskSummary[];
  
  // 评语
  comments: BossComment[];
}

type EmployeeStatus = 
  | 'idle'             // 空闲
  | 'working'          // 工作中
  | 'waiting'          // 待命（等待前置任务）
  | 'unavailable';    // 不可用
```

### 2.5 偏好理解 (PreferenceUnderstanding)

```typescript
interface PreferenceUnderstanding {
  // 显式偏好（老板主动设置）
  explicit: Record<string, string>;  // { "视频时长": "30秒", "风格": "科技感" }
  
  // 学习到的偏好
  learned: LearnedPreference[];
  
  // 偏好更新历史
  history: PreferenceHistory[];
}

interface LearnedPreference {
  tag: string;                     // 标签：节奏/转场/配乐/字幕/封面/...
  value: string;                   // 偏好值：快/慢/平滑/电子/无边框/...
  confidence: number;              // 置信度 0-1
  
  // 来源
  evidence: PreferenceEvidence[];
  
  // 更新时间
  updated_at: string;
  update_count: number;            // 更新次数
}

interface PreferenceEvidence {
  project_id: string;
  task_id: string;
  comment: string;                  // 原始评语
  extracted_at: string;
}

interface PreferenceHistory {
  tag: string;
  old_value?: string;
  new_value: string;
  source_comment: string;
  changed_at: string;
}
```

### 2.6 老板评语 (BossComment)

```typescript
interface BossComment {
  id: string;
  
  // 关联
  project_id: string;
  task_id: string;
  employee_role: EmployeeRole;
  
  // 内容
  content: string;                  // 原始评语
  sentiment: 'positive' | 'negative' | 'neutral';  // 情感
  
  // 提取
  extracted_tags: string[];         // 提取的偏好标签
  action_items?: string[];        // 行动项
  
  // 时间
  created_at: string;
}
```

### 2.7 工作流 (Workflow)

```typescript
interface Workflow {
  id: string;
  name: string;
  project_type: ProjectType;
  
  // 步骤定义
  steps: WorkflowStep[];
}

interface WorkflowStep {
  order: number;
  employee_role: EmployeeRole;
  name: string;
  description: string;
  
  // 输入
  input_from_previous: boolean;    // 是否接收上一步输出
  additional_input?: string;       // 额外输入要求
  
  // 技能
  required_skills?: string[];       // 需要的技能
  
  // 输出
  expected_output: string;          // 期望输出描述
  
  // 条件
  condition?: string;              // 执行条件
}
```

### 2.8 交付物 (Deliverable)

```typescript
interface Deliverable {
  id: string;
  project_id: string;
  task_id: string;
  
  // 类型
  type: 'video' | 'image' | 'text' | 'audio';
  
  // 内容
  url: string;                     // 文件路径/URL
  description?: string;
  
  // 验收
  approved: boolean;
  approved_at?: string;
  boss_comment?: string;
  
  // 时间
  created_at: string;
}
```

## 四、存储设计

### 4.1 Memory命名空间

```typescript
// 项目存储
namespace: 'projects'
key: project.id  // e.g., "proj_20260305_001"

// 员工档案存储
namespace: 'employees'
key: employee.role  // e.g., "editor"

// 老板偏好存储
namespace: 'boss_preferences'
key: 'default'  // 当前老板

// 评语历史存储
namespace: 'comments'
key: comment.id
```

### 4.2 检索示例

```javascript
// 获取某员工的所有评语
await memory.search({
  namespace: 'comments',
  filter: {
    employee_role: 'editor'
  },
  sort: 'created_at',
  limit: 20
});

// 获取某项目的所有任务
await memory.read({
  namespace: 'projects',
  key: 'proj_20260305_001'
}).then(p => p.tasks);
```
