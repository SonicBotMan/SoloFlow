# 🧠 偏好学习算法

## 一、概述

偏好学习是AI一人公司的核心特性之一。系统会自动从老板的反馈中提取偏好标签，并更新员工对老板偏好的理解。

### 2.2 提取算法

```javascript
class PreferenceExtractor {
  
  // 从评语中提取偏好标签
  static extractTags(comment) {
    const tags = [];
    
    for (const [category, config] of Object.entries(PREFERENCE_PATTERNS)) {
      const match = comment.match(config.pattern);
      if (match) {
        const rawValue = match[1];
        const normalizedValue = config.mapping[rawValue] || rawValue;
        tags.push({
          category: category,
          value: normalizedValue,
          original: rawValue
        });
      }
    }
    
    return tags;
  }
  
  // 提取行动项
  static extractActionItems(comment) {
    const actionPatterns = [
      /下次(.+)/,
      /以后(.+)/,
      /再用(.+)/,
      /改用(.+)/,
      /不要(.+)/
    ];
    
    const actions = [];
    for (const pattern of actionPatterns) {
      const match = comment.match(pattern);
      if (match) {
        actions.push(match[0]);
      }
    }
    
    return actions;
  }
  
  // 判断情感
  static extractSentiment(comment) {
    const negativeWords = ['太', '慢', '差', '烂', '生硬', '难听', '不好看'];
    const positiveWords = ['好', '棒', '不错', '喜欢', '流畅', '好看'];
    
    for (const word of negativeWords) {
      if (comment.includes(word)) {
        return 'negative';
      }
    }
    
    for (const word of positiveWords) {
      if (comment.includes(word)) {
        return 'positive';
      }
    }
    
    return 'neutral';
  }
}
```

## 三、置信度计算

### 3.1 初始置信度

```javascript
const INITIAL_CONFIDENCE = 0.6;  // 首次反馈
const MIN_CONFIDENCE = 0.2;       // 最低置信度
const MAX_CONFIDENCE = 0.95;      // 最高置信度
```

### 3.2 置信度更新规则

```javascript
class ConfidenceCalculator {
  
  // 更新置信度
  static updateConfidence(preference, newValue, sentiment) {
    if (preference.value === newValue) {
      // 相同偏好，增加置信度
      const increment = sentiment === 'positive' ? 0.15 : 0.1;
      preference.confidence = Math.min(
        MAX_CONFIDENCE, 
        preference.confidence + increment
      );
      preference.update_count++;
    } else {
      // 偏好改变
      if (sentiment === 'negative') {
        // 负反馈：降低旧偏好置信度，设置新偏好
        preference.value = newValue;
        preference.confidence = INITIAL_CONFIDENCE;
        preference.update_count = 1;
      } else if (sentiment === 'positive') {
        // 正反馈：新偏好添加为额外偏好
        // 不修改原有偏好
      }
    }
    
    preference.updated_at = new Date().toISOString();
    return preference;
  }
  
  // 基于证据数量计算置信度
  static calculateFromEvidence(evidenceCount) {
    // 证据越多，置信度越高
    const base = 0.3;
    const increment = 0.1;
    return Math.min(
      MAX_CONFIDENCE,
      base + (evidenceCount * increment)
    );
  }
}
```

### 3.3 置信度等级

```javascript
const CONFIDENCE_LEVELS = {
  HIGH: 0.8,      // 高置信度 - 非常确定
  MEDIUM: 0.5,   // 中置信度 - 比较确定
  LOW: 0.3       // 低置信度 - 初步判断
};

function getConfidenceLevel(confidence) {
  if (confidence >= CONFIDENCE_LEVELS.HIGH) return '高';
  if (confidence >= CONFIDENCE_LEVELS.MEDIUM) return '中';
  return '低';
}
```

## 四、偏好更新

### 4.1 更新流程

```javascript
class PreferenceEngine {
  
  // 记录反馈并更新偏好
  static async recordFeedback(feedback) {
    const { project_id, task_id, employee_role, content, sentiment } = feedback;
    
    // 1. 提取标签
    const extractedTags = PreferenceExtractor.extractTags(content);
    const actionItems = PreferenceExtractor.extractActionItems(content);
    
    // 2. 创建评语记录
    const comment = {
      id: `comment_${Date.now()}`,
      project_id,
      task_id,
      employee_role,
      content,
      sentiment,
      extracted_tags: extractedTags.map(t => `${t.category}=${t.value}`),
      action_items: actionItems,
      created_at: new Date().toISOString()
    };
    
    // 3. 保存评语
    await memory.write({
      namespace: 'comments',
      key: comment.id,
      value: comment
    });
    
    // 4. 更新员工偏好
    for (const tag of extractedTags) {
      await this.updateEmployeePreference(
        employee_role,
        tag.category,
        tag.value,
        sentiment,
        comment.id
      );
    }
    
    return comment;
  }
  
  // 更新单个偏好
  static async updateEmployeePreference(
    employeeRole, 
    category, 
    value, 
    sentiment,
    commentId
  ) {
    // 读取当前员工档案
    const employee = await memory.read({
      namespace: 'employees',
      key: employeeRole
    });
    
    // 初始化偏好结构
    if (!employee.preference_understanding.learned) {
      employee.preference_understanding.learned = {};
    }
    
    // 获取或创建偏好
    let preference = employee.preference_understanding.learned[category];
    
    if (!preference) {
      preference = {
        tag: category,
        value: value,
        confidence: INITIAL_CONFIDENCE,
        evidence: [],
        updated_at: new Date().toISOString(),
        update_count: 0
      };
    }
    
    // 更新置信度
    preference = ConfidenceCalculator.updateConfidence(
      preference, 
      value, 
      sentiment
    );
    
    // 添加证据
    preference.evidence.push({
      project_id: commentId.split('_')[1], // 简化处理
      comment_id: commentId,
      extracted_at: new Date().toISOString()
    });
    
    // 保存历史
    employee.preference_understanding.history.push({
      tag: category,
      old_value: preference.value,
      new_value: value,
      source_comment: commentId,
      changed_at: new Date().toISOString()
    });
    
    // 写回存储
    employee.preference_understanding.learned[category] = preference;
    
    await memory.write({
      namespace: 'employees',
      key: employeeRole,
      value: employee
    });
  }
}
```

## 五、偏好应用

### 5.1 应用偏好到任务

```javascript
class PreferenceApplier {
  
  // 获取员工对当前任务的偏好
  static async getTaskPreferences(employeeRole, taskContext) {
    const employee = await memory.read({
      namespace: 'employees',
      key: employeeRole
    });
    
    const preferences = employee.preference_understanding.learned;
    const applied = [];
    
    for (const [category, pref] of Object.entries(preferences)) {
      if (pref.confidence >= CONFIDENCE_LEVELS.LOW) {
        applied.push({
          category: category,
          value: pref.value,
          confidence: pref.confidence,
          from_evidence: pref.evidence.length
        });
      }
    }
    
    return applied;
  }
  
  // 生成偏好应用说明
  static generatePreferenceNote(preferences) {
    if (!preferences || preferences.length === 0) {
      return '';
    }
    
    const lines = ['**使用的老板偏好:**'];
    
    for (const pref of preferences) {
      const emoji = pref.confidence >= CONFIDENCE_LEVELS.HIGH ? '✓✓' : '✓';
      lines.push(`- ${emoji} ${pref.category}: ${pref.value}`);
    }
    
    return lines.join('\n');
  }
}
```

### 6.1 生成员工偏好报告

```javascript
function generateEmployeePreferenceReport(employee) {
  const prefs = employee.preference_understanding.learned;
  const history = employee.preference_understanding.history;
  
  // 按置信度排序
  const sortedPrefs = Object.values(prefs)
    .filter(p => p.confidence >= 0.3)
    .sort((a, b) => b.confidence - a.confidence);
  
  // 格式化输出
  const lines = [
    `## 🧠 对老板偏好的理解`,
    '',
    `| 偏好项 | 偏好值 | 置信度 | 证据数 |`,
    `|--------|--------|--------|--------|`
  ];
  
  for (const pref of sortedPrefs) {
    const level = getConfidenceLevel(pref.confidence);
    const stars = '⭐'.repeat(Math.ceil(pref.confidence * 5));
    lines.push(
      `| ${pref.tag} | ${pref.value} | ${stars} ${Math.round(pref.confidence*100)}% | ${pref.evidence.length} |`
    );
  }
  
  // 最近评语
  const recentComments = employee.comments?.slice(-3) || [];
  if (recentComments.length > 0) {
    lines.push('', '### 💬 最近评语', '');
    for (const comment of recentComments) {
      lines.push(`- "${comment.content}"`);
    }
  }
  
  return lines.join('\n');
}
```
